    view for navigation element, currently just adds button for navigating back to library

        Views.NavHelper = Marionette.ItemView.extend(
      template: navHelper
      events:
        "click button.js-library-back": "navigateToLibrary"

      initialize: (options) ->
        options = options or {}
        @buttonText = options.buttonText or "Back"

      serializeData: ->
        buttonText: @buttonText

      onBeforeClose: ->
        @$el.transition
          perspective: "100px"
          rotate3d: "1,1,0,180deg"

      navigateToLibrary: ->
        @$el.transition
          perspective: "100px"
          rotate3d: "1,1,0,180deg"
        Swabcast.trigger "episodes:library"
        @close()
    )
