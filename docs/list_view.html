      onRender: ->
        @$el.addClass "disabled"  if @model.get("enqueue") is true

      editClicked: (e) ->
        e.preventDefault()
        e.stopPropagation()
        @trigger "subscription:edit", @model

      highlightName: ->
        @$el.toggleClass "success"

      toggleQueue: (e) ->
        e.preventDefault()
        e.stopPropagation()
        @$el.addClass "disabled"
        if @model.get("enqueue") is false
          @model.set enqueue: true
